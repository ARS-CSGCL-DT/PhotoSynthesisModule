328  void CGasExchange::EnergyBalance()
343  {
344  const long Lambda = 44000; //latent heat of vaporization of water J mol-1 - not used in this implementation
345  const double Cp = 29.3; // thermodynamic psychrometer constant and specific hear of air, J mol-1 C-1
346  const double psc = 6.66e-4; //psycrometric constant units are C-1
347  //psc=Cp/Lambda = 29.3/44000 See Campbell and Norman, pg 232, after eq 14.11
348 
349  //The following are secondary variables used in the energy balance
350  double HeatConductance, //heat conductance J m-2 s-1
351  VaporConductance, //vapor conductance ratio of stomatal and heat conductance mol m-2 s-1
352  RadiativeConductance, //radiative conductance J m-2 s-1
353  RadiativeAndHeatConductance, //radiative+heat conductance
354  psc1, // apparent psychrometer constant Campbell and Norman, page 232 after eq 14.11
355  Ea, //ambient vapor pressure kPa
356  thermal_air; // emitted thermal radiation Watts m-2
357  double lastTi, newTi;
358  int iter;
359 
360  HeatConductance = BoundaryLayerConductance*(0.135/0.147); // heat conductance, HeatConductance = 1.4*.135*sqrt(u/d), u is the wind speed in m/s} Boundary Layer Conductance to Heat
361  // Since BoundaryLayerConductance is .147*sqrt(u/d) this scales to 0.135*sqrt(u/d) - HeatConductance on page 109 of Campbell and Norman, 1998
362  // Wind was accounted for in BoundaryLayerConductance already as BoundaryLayerConductance (turbulent vapor transfer) was calculated from CalcTurbulentVaporConductance() in GasEx.
363  // units are J m-2 s-1
364  VaporConductance = StomatalConductance*BoundaryLayerConductance/(StomatalConductance+BoundaryLayerConductance); //vapor conductance, StomatalConductance is stomatal conductance and is given as gvs in Campbell and Norman.
365  RadiativeConductance = (4*epsilon*sbc*pow(273+Tair,3)/Cp)*2; // radiative conductance, *2 account for both sides
366  RadiativeAndHeatConductance = HeatConductance + RadiativeConductance;
367  thermal_air = epsilon*sbc*pow(Tair+273,4)*2; //Multiply by 2 for both surfaces
368  psc1 = psc*RadiativeAndHeatConductance/VaporConductance;
369  this->VPD = Es(Tair)*(1-RH); // vapor pressure deficit Es is saturation vapor pressure at air temperature
370  // iterative version
371  newTi=-10;
372  iter=0;
373  lastTi=Tleaf;
374  double Res, dRes; //temporary variables
375  double thermal_leaf;
376  Ea = Es(Tair)*RH; // ambient vapor pressure
377  while ((abs(lastTi-newTi)>0.001) && (iter <maxiter))
378  {
379  lastTi=newTi;
380  Tleaf= Tair + (R_abs- thermal_air-Lambda*VaporConductance*this->VPD/Press)/(Cp*RadiativeAndHeatConductance+Lambda*Slope(Tair)*VaporConductance); // eqn 14.6a
381  thermal_leaf=epsilon*sbc*pow(Tleaf+273,4)*2;
382  Res = R_abs - thermal_leaf - Cp*HeatConductance*(Tleaf - Tair) - Lambda*VaporConductance*0.5*(Es(Tleaf)-Ea)/Press; // Residual function: f(Ti), KT Paw (1987)
383  dRes= -4*epsilon*sbc*pow(273+Tleaf,3)*2-Cp*HeatConductance*Tleaf-Lambda*VaporConductance*Slope(Tleaf); // derivative of residual: f'(Ti)
384  newTi = Tleaf + Res/dRes; // newton-rhapson iteration
385  iter++;
386  }
387  Tleaf=newTi;
388 
389  Transpiration =1000*VaporConductance*(Es(Tleaf)-Ea)/Press; //Don't need Lambda - cancels out see eq 14.10 in Campbell and Norman, 1998
390  // umol m-2 s-1. note 1000 converts from moles to umol
391  }
